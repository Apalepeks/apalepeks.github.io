let x, y; function setup() { createCanvas(800, 600); background(255); // Start in the center of the canvas x = width / 2; y = height / 2; stroke(0); // Set the stroke color to black strokeWeight(2); // Set the stroke weight } function draw() { // Draw a point at the current position point(x, y); // Move in a random direction let stepSize = 2; let r = floor(random(4)); switch (r) { case 0: x += stepSize; break; case 1: x -= stepSize; break; case 2: y += stepSize; break; case 3: y -= stepSize; break; } // Keep the walker within the bounds of the canvas x = constrain(x, 0, width - 1); y = constrain(y, 0, height - 1); }
